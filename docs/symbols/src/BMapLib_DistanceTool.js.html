332             var range,begin;

405         /**
406          * 为目标元素添加事件监听器
407          * @name baidu.event.on
408          * @function
409          * @grammar baidu.event.on(element, type, listener)
410          * @param {HTMLElement|string|window} element 目标元素或目标元素id
411          * @param {string} type 事件类型
412          * @param {Function} listener 需要添加的监听器
413          * @remark
414          *  1. 不支持跨浏览器的鼠标滚轮事件监听器添加<br>
415          *  2. 改方法不为监听器灌入事件对象，以防止跨iframe事件挂载的事件对象获取失败            
416          * @shortcut on
417          * @meta standard
418          * @see baidu.event.un
419          *             
420          * @returns {HTMLElement|window} 目标元素
421          */
422         baidu.on = baidu.event.on = function (element, type, listener) {
423             type = type.replace(/^on/i, '');
424             element = baidu._g(element);
425             var realListener = function (ev) {
426                 // 1. 这里不支持EventArgument,  原因是跨frame的事件挂载
427                 // 2. element是为了修正this
428                 listener.call(element, ev);
429             },
430             lis = baidu.event._listeners,
431             filter = baidu.event._eventFilter,
432             afterFilter,
433             realType = type;
434             type = type.toLowerCase();
435             // filter过滤
436             if(filter && filter[type]){
437                 afterFilter = filter[type](element, type, realListener);
438                 realType = afterFilter.type;
439                 realListener = afterFilter.listener;
440             }
441             // 事件监听器挂载
442             if (element.addEventListener) {
443                 element.addEventListener(realType, realListener, false);
444             } else if (element.attachEvent) {
445                 element.attachEvent('on' + realType, realListener);
446             }
447           
448             // 将监听器存储到数组中
449             lis[lis.length] = [element, type, listener, realListener, realType];
450             return element;
451         };

453         /**
454          * 为目标元素移除事件监听器
455          * @name baidu.event.un
456          * @function
457          * @grammar baidu.event.un(element, type, listener)
458          * @param {HTMLElement|string|window} element 目标元素或目标元素id
459          * @param {string} type 事件类型
460          * @param {Function} listener 需要移除的监听器

462          * @meta standard
463          *             
464          * @returns {HTMLElement|window} 目标元素

466         baidu.un = baidu.event.un = function (element, type, listener) {
467             element = baidu._g(element);
468             type = type.replace(/^on/i, '').toLowerCase();
469             
470             var lis = baidu.event._listeners, 
471                 len = lis.length,
472                 isRemoveAll = !listener,
473                 item,
474                 realType, realListener;
475             
476             //如果将listener的结构改成json
477             //可以节省掉这个循环，优化性能
478             //但是由于un的使用频率并不高，同时在listener不多的时候
479             //遍历数组的性能消耗不会对代码产生影响
480             //暂不考虑此优化
481             while (len--) {
482                 item = lis[len];
483                 
484                 // listener存在时，移除element的所有以listener监听的type类型事件
485                 // listener不存在时，移除element的所有type类型事件
486                 if (item[1] === type
487                     && item[0] === element
488                     && (isRemoveAll || item[2] === listener)) {
489                     realType = item[4];
490                     realListener = item[3];
491                     if (element.removeEventListener) {
492                         element.removeEventListener(realType, realListener, false);
493                     } else if (element.detachEvent) {
494                         element.detachEvent('on' + realType, realListener);
495                     }
496                     lis.splice(len, 1);
497                 }
498             }            
499             return element;
500         };
